#Základní příkazy

-----

#Příkaz                                                       #Vysvětlení                                                      #Příklad output

ls (list)                                                     vypíše obsah aktuální složky                                      Videos  Photos Work
cd (change directory)                                         změní adresář
cd .. (change directory)                                      skočí o úroveň adresáře zpět
pwd (print working directory)                                 vypíše aktuální pracovní místo (kde se právě nacházíš)            /user/mnt/usb/kopie
cat (concatenate)  SOUBOR.txt                                 vypíše obsah souboru SOUBOR.txt
touch SOUBOT.txt                                              vytvoří prázdný soubor SOUBOR.txt
nano SOUBOR.txt                                               otevře SOUBOR.txt v jednoduchém textové editoru (pokud ještě neexistoval i ho vytvoří)
cp /cesta-k-souboru /cil-souboru                              nakopiruje soubor/složku z /cesta-k-souboru do /cil-souboru
mv /cesta-k-souboru /cil-souboru                              přesune soubor/složku z /cesta-k-souboru do /cil-souboru
rm SOUBOR.txt                                                 odstraní SOUBOR.txt
mkdir NAZEV-SLOZKY                                            vytvoří složku NAZEV-SLOZKY
rmdir NAZEV-SLOZKY                                            odstraní složku NAZEV-SLOZKY
echo "Hello world!"                                           vypíše do terminálu Hello world!                                    Hello world!
sudo                                                          pokud dáš sudo před příkaz provede se jako root (všechna práva)
sudo apt install *nazev aplikace*                             nainstaluje aplikaci
